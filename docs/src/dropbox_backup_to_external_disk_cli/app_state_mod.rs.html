9/// AppState is used as a global variable/struct.
10/// AppState struct contains only private fields. Some are immutable and other are mutable behind a Mutex.
11/// The struct must be declared inside the bin project. Because only that way I can add the AppStateMethods from the LIB project. (Rust Orphan rule)
12/// These methods from AppStateMethods are "dependency injection" or "inversion of control". Inside the LIB project these methods are declared and used, but there is no code.
13/// The code is inside the bin project, because different bin projects can bring different methods implementations.
14#[derive(Debug)]
15struct AppState {
16    // immutable
17    app_config: lib::AppConfig,
18    // mutable with Mutex
19    string_proba_mutex: std::sync::Mutex<String>,
20}
21
22/// implementation of AppStateMethods functions that is defined in the lib project
23/// and will be used in the lib project, because I want the lib project to have no idea where the tokens are stored.
24impl lib::AppStateMethods for AppState {
25    fn load_keys_from_io(&self) -> lib::Result<(String, String)> {
26        let master_key = std::env::var("DBX_KEY_1")?;
27        let token_enc = std::env::var("DBX_KEY_2")?;
28        Ok((master_key, token_enc))
29    }
30    fn ref_app_config(&self) -> &lib::AppConfig {
31        &self.app_config
32    }
33    fn lock_proba(&'_ self) -> MutexGuard<'_, String> {
34        self.string_proba_mutex.lock().unwrap()
35    }
36}
37
